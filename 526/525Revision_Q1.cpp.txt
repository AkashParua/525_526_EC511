#include <iostream>
#include <climits>
using namespace std;

int main() {
    
    int n = 8;
    
    int *arr = new int[n];
    
    cout<<"Enter 8 elements in the array: ";
    for(int i=0;i<n;i++){
        int k;
        cin>>k;
        arr[i] = k;
    }
    
    int largest = INT_MIN;
    int smallest = INT_MAX;
    
    for(int i=0;i<n;i++){
        if(arr[i] > largest)
        largest = arr[i];
        
        if(arr[i] < smallest)
        smallest = arr[i];
    }
    
    cout<<"Largest number: "<<largest<<endl;
    cout<<"Smallest number: "<<smallest;
    

    return 0;
}
